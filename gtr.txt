        [Test]
        public void SplitDataAcrossSheets_Creates3SheetsFor1MillionRows()
        {
            // Arrange
            var data = GenerateTestDataWithIds(1_000_000);

            // Act
            _largeColumnSheetBuilder.SplitDataAcrossSheets(data);

            // Assert
            // The first sheet's data is set on the original configuration
            var originalConfig = GetConfiguration(_largeColumnSheetBuilder);
            Assert.That(originalConfig.Data.Count, Is.EqualTo(400_000));
            
            // Additional sheets are added to the file builder
            var addedSheets = GetAddedSheets();

        [Test]
        public void SplitDataAcrossSheets_CorrectlyNamesSheets()
        {
            // Arrange
            var data = GenerateTestDataWithIds(800_001);

            // Act
            _largeColumnSheetBuilder.SplitDataAcrossSheets(data);

            // Assert
            var originalConfig = GetConfiguration(_largeColumnSheetBuilder);
            Assert.That(originalConfig.SheetName, Is.EqualTo("LargeSheet"));
            
            var addedSheets = GetAddedSheets();
            var sheetNames = addedSheets.Select(s => s.SheetName).ToList();
            Assert.That(sheetNames, Contains.Item("LargeSheet 1"));
            Assert.That(sheetNames, Contains.Item("LargeSheet 2"));
        }

        [Test]
        public void SplitDataAcrossSheets_PreservesDataOrder()
        {
            // Arrange
            var data = new List<LargeColumnModel>();
            for (int i = 1; i <= 800_001; i++)
            {
                data.Add(new LargeColumnModel { Column1 = $"Row{i}" });
            }

            // Act
            _largeColumnSheetBuilder.SplitDataAcrossSheets(data);

            // Assert
            var originalConfig = GetConfiguration(_largeColumnSheetBuilder);
            Assert.That(originalConfig.Data.First().Column1, Is.EqualTo("Row1"));
            Assert.That(originalConfig.Data.Last().Column1, Is.EqualTo("Row400000"));

            var addedSheets = GetAddedSheets();
            var sheet1Data = ((SheetConfiguration<LargeColumnModel>)addedSheets[addedSheets.Count - 2]).Data;
            var sheet2Data = ((SheetConfiguration<LargeColumnModel>)addedSheets[addedSheets.Count - 1]).Data;
            
            Assert.That(sheet1Data.First().Column1, Is.EqualTo("Row400001"));
            Assert.That(sheet1Data.Last().Column1, Is.EqualTo("Row800000"));
            Assert.That(sheet2Data.First().Column1, Is.EqualTo("Row800001"));
        }

        [Test]
        public void CloneConfiguration_CreatesNewInstanceWithCorrectName()
        {
            // Arrange
            SetupOriginalConfiguration();

            // Act
            var cloned = _largeColumnSheetBuilder.CloneConfiguration("NewSheet");

            // Assert
            Assert.That(cloned, Is.Not.Null);
            Assert.That(cloned.SheetName, Is.EqualTo("NewSheet"));
            Assert.That(cloned, Is.Not.SameAs(GetConfiguration(_largeColumnSheetBuilder)));
        }

        [Test]
        public void CloneConfiguration_CopiesDropdownData()
        {
            // Arrange
            var dropdownData = new List<ExcelDropdownData>
            {
                new ExcelDropdownData
                {
                    ColumnName = "Status",
                    DataList = new List<object> { "Active", "Inactive", "Pending" },
                    BindProperties = new List<string> { "Column1" }
                }
            };
            _largeColumnSheetBuilder.WithDropdownData(dropdownData, "DropdownSheet");

            // Act
            var cloned = _largeColumnSheetBuilder.CloneConfiguration("ClonedSheet");

            // Assert
            Assert.That(cloned.DropdownData, Is.Not.Null);
            Assert.That(cloned.DropdownData.Count, Is.EqualTo(1));
            Assert.That(cloned.DropdownData[0].ColumnName, Is.EqualTo("Status"));
            Assert.That(cloned.DropdownSheetName, Is.EqualTo("DropdownSheet"));
        }

        [Test]
        public void CloneConfiguration_CopiesConditionalRules()
        {
            // Arrange
            var rules = new List<ExcelConditionalRule>
            {
                new ExcelConditionalRule
                {
                    SourcePropertyName = "Column1",
                    TargetPropertyName = "Column2",
                    Operator = ConditionalOperator.Equal,
                    Values = new List<string> { "TestValue" },
                    Actions = new List<ConditionalAction>
                    {
                        new ConditionalAction
                        {
                            Type = ConditionalActionType.ChangeBackgroundColor,
                            Value = "#FF0000"
                        }
                    }
                }
            };
            
            var conditionalFactory = new TestConditionalRulesFactory<LargeColumnModel>(rules);
            _largeColumnSheetBuilder.WithConditionalRules(conditionalFactory);

            // Act
            var cloned = _largeColumnSheetBuilder.CloneConfiguration("ClonedSheet");

            // Assert
            Assert.That(cloned.ConditionalRules, Is.Not.Null);
            Assert.That(cloned.ConditionalRules.Count, Is.EqualTo(1));
            Assert.That(cloned.ConditionalRules[0].SourcePropertyName, Is.EqualTo("Column1"));
        }

        [Test]
        public void CloneConfiguration_CopiesExcludedProperties()
        {
            // Arrange
            _largeColumnSheetBuilder.ExcludeProperties(
                x => x.Column1,
                x => x.Column2,
                x => x.Column3
            );

            // Act
            var cloned = _largeColumnSheetBuilder.CloneConfiguration("ClonedSheet");

            // Assert
            Assert.That(cloned.ExcludedProperties, Is.Not.Null);
            Assert.That(cloned.ExcludedProperties.Count, Is.EqualTo(3));
            Assert.That(cloned.ExcludedProperties, Contains.Item("Column1"));
            Assert.That(cloned.ExcludedProperties, Contains.Item("Column2"));
            Assert.That(cloned.ExcludedProperties, Contains.Item("Column3"));
        }

        [Test]
        public void CloneConfiguration_DoesNotCopyData()
        {
            // Arrange
            var originalData = GenerateTestData<LargeColumnModel>(100);
            _largeColumnSheetBuilder.WithData(originalData);

            // Act
            var cloned = _largeColumnSheetBuilder.CloneConfiguration("ClonedSheet");

            // Assert
            Assert.That(cloned.Data, Is.Not.Null);
            Assert.That(cloned.Data.Count, Is.EqualTo(0), "Cloned configuration should start with empty data");
        }

        [Test]
        public void CloneConfiguration_IndependentExcludedProperties()
        {
            // Arrange
            _largeColumnSheetBuilder.ExcludeProperties(x => x.Column1);
            var cloned = _largeColumnSheetBuilder.CloneConfiguration("ClonedSheet");

            // Act - Modify cloned excluded properties
            cloned.ExcludedProperties.Add("Column99");

            // Assert - Original should not be affected
            var original = GetConfiguration(_largeColumnSheetBuilder);
            Assert.That(original.ExcludedProperties, Does.Not.Contain("Column99"));
            Assert.That(cloned.ExcludedProperties, Contains.Item("Column99"));
        }

        #region Helper Methods

        private List<T> GenerateTestData<T>(int count) where T : new()
        {
            var data = new List<T>();
            for (int i = 0; i < count; i++)
            {
                data.Add(new T());
            }
            return data;
        }

        private List<LargeColumnModel> GenerateTestDataWithIds(int count)
        {
            var data = new List<LargeColumnModel>();
            for (int i = 1; i <= count; i++)
            {
                data.Add(new LargeColumnModel { Column1 = i.ToString() });
            }
            return data;
        }

        private SheetConfiguration<T> GetConfiguration<T>(SheetBuilder<T> builder) where T : class, new()
        {
            // Access the private _configuration field
            var field = typeof(SheetBuilder<T>).GetField("_configuration", 
                BindingFlags.NonPublic | BindingFlags.Instance);
            return (SheetConfiguration<T>)field.GetValue(builder);
        }

        private List<ISheetConfiguration> GetAddedSheets()
        {
            // Access the private _sheets field from ExcelFileBuilder
            var field = typeof(ExcelFileBuilder).GetField("_sheets", 
                BindingFlags.NonPublic | BindingFlags.Instance);
            return (List<ISheetConfiguration>)field.GetValue(_fileBuilder);
        }

        private void SetupOriginalConfiguration()
        {
            var config = GetConfiguration(_largeColumnSheetBuilder);
            config.DropdownData = new List<ExcelDropdownData>();
            config.ConditionalRules = new List<ExcelConditionalRule>();
        }

        private object GetPropertyByName<T>(T obj, string propertyName)
        {
            var property = typeof(T).GetProperty(propertyName);
            return property?.GetValue(obj);
        }

        private class TestConditionalRulesFactory<T> : IExcelConditionalRulesFactory<T> where T : class
        {
            public List<ExcelConditionalRule> Rules { get; }

            public TestConditionalRulesFactory(List<ExcelConditionalRule> rules)
            {
                Rules = rules;
            }
        }

        #endregion
