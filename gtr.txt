[TestFixture]
public class SheetBuilderTests
{
    private const string TestSheetName = "TestSheet";
    private const int MaxRowsPerSheet = 400000;
    private const int LargeDataSetSize = 900000;
    private const int MediumDataSetSize = 300000;
    private const int SmallDataSetSize = 100000;
    private const int TestDataMultiplier = 10;
    private const int ColumnThreshold = 50;
    
    private Mock<ExcelFileBuilder> _mockFileBuilder;
    private TestableSheetBuilder<TestModel> _sheetBuilder;

    [SetUp]
    public void Setup()
    {
        var mockCacheManager = new Mock<IExcelCacheManager>();
        var mockCellWriter = new Mock<IExcelCellWriter>();
        var mockStyleManager = new Mock<IExcelStyleManager>();
        var mockValidationManager = new Mock<IExcelValidationManager>();
        var stringBuilderPool = new StringBuilderPool();
        var mockPropertyNameFormatter = new Mock<IPropertyNameFormatter>();
        var mockWorkbookWriter = new Mock<IExcelWorkbookWriter>();

        _mockFileBuilder = new Mock<ExcelFileBuilder>(
            mockCacheManager.Object,
            mockCellWriter.Object,
            mockStyleManager.Object,
            mockValidationManager.Object,
            stringBuilderPool,
            mockPropertyNameFormatter.Object,
            mockWorkbookWriter.Object
        );

        _sheetBuilder = new TestableSheetBuilder<TestModel>(_mockFileBuilder.Object, TestSheetName);
    }

    [Test]
    public void ShouldSplitData_WhenDataCountBelowMaxRows_ReturnsFalse()
    {
        var data = GenerateTestData(SmallDataSetSize);

        var result = _sheetBuilder.TestShouldSplitData(data);

        Assert.That(result, Is.False, "Should not split when data count is below max rows");
    }

    [Test]
    public void ShouldSplitData_WhenDataCountExceedsMaxRowsAndFewColumns_ReturnsFalse()
    {
        var data = GenerateTestData(MaxRowsPerSheet + 1);

        var result = _sheetBuilder.TestShouldSplitData(data);

        Assert.That(result, Is.False, "Should not split when column count is below threshold");
    }

    [Test]
    public void ShouldSplitData_WhenDataCountExceedsMaxRowsAndManyColumns_ReturnsTrue()
    {
        var sheetBuilder = new TestableSheetBuilder<LargeTestModel>(_mockFileBuilder.Object, TestSheetName);
        var data = GenerateLargeTestData(MaxRowsPerSheet + 1);

        var result = sheetBuilder.TestShouldSplitData(data);

        Assert.That(result, Is.True, "Should split when both row count exceeds max and columns exceed threshold");
    }

    [Test]
    public void ShouldSplitData_WhenDataCountExactlyMaxRows_ReturnsFalse()
    {
        var data = GenerateTestData(MaxRowsPerSheet);

        var result = _sheetBuilder.TestShouldSplitData(data);

        Assert.That(result, Is.False, "Should not split when data count equals max rows");
    }

    [Test]
    public void SplitDataAcrossSheets_WithSingleSheetWorthOfData_SetsDataOnMainConfiguration()
    {
        var data = GenerateTestData(MediumDataSetSize);

        _sheetBuilder.TestSplitDataAcrossSheets(data);
        var configuration = _sheetBuilder.GetConfiguration();

        Assert.That(configuration.Data, Has.Count.EqualTo(MediumDataSetSize));
        Assert.That(configuration.Data, Is.EqualTo(data.Take(MediumDataSetSize)));
    }

    [Test]
    public void SplitDataAcrossSheets_WithMultipleSheetsWorthOfData_SetsFirstSheetData()
    {
        var data = GenerateTestData(LargeDataSetSize);
        
        var realFileBuilder = CreateRealFileBuilder();
        var realSheetBuilder = new TestableSheetBuilder<TestModel>(realFileBuilder, TestSheetName);

        realSheetBuilder.TestSplitDataAcrossSheets(data);
        var mainConfiguration = realSheetBuilder.GetConfiguration();

        Assert.That(mainConfiguration.Data, Has.Count.EqualTo(MaxRowsPerSheet), "Main configuration should have max rows");
        Assert.That(mainConfiguration.Data[0].Id, Is.EqualTo(1), "First item should be from start of data");
        Assert.That(mainConfiguration.Data[MaxRowsPerSheet - 1].Id, Is.EqualTo(MaxRowsPerSheet), "Last item should be at position 400000");
    }

    [Test]
    public void SplitDataAcrossSheets_PreservesDataInMainConfiguration()
    {
        var data = GenerateTestData(LargeDataSetSize);
        
        _sheetBuilder.TestSplitDataAcrossSheets(data);
        var mainConfiguration = _sheetBuilder.GetConfiguration();

        Assert.That(mainConfiguration.Data, Has.Count.EqualTo(MaxRowsPerSheet));
        
        for (int i = 0; i < MaxRowsPerSheet; i++)
        {
            Assert.That(mainConfiguration.Data[i].Id, Is.EqualTo(data[i].Id), $"Data mismatch at index {i}");
        }
    }

    [Test]
    public void SplitDataAcrossSheets_WithExactMultipleOfMaxRows_SetsCorrectDataInMainConfiguration()
    {
        var dataSize = MaxRowsPerSheet * 2;
        var data = GenerateTestData(dataSize);
        
        _sheetBuilder.TestSplitDataAcrossSheets(data);
        var mainConfiguration = _sheetBuilder.GetConfiguration();

        Assert.That(mainConfiguration.Data, Has.Count.EqualTo(MaxRowsPerSheet));
        Assert.That(mainConfiguration.Data.First().Id, Is.EqualTo(1));
        Assert.That(mainConfiguration.Data.Last().Id, Is.EqualTo(MaxRowsPerSheet));
    }

    [Test]
    public void SplitDataAcrossSheets_CallsAddSheetConfigurationForAdditionalSheets()
    {
        var data = GenerateTestData(LargeDataSetSize);
        var realFileBuilder = CreateRealFileBuilder();
        var sheetBuilder = new TestableSheetBuilder<TestModel>(realFileBuilder, TestSheetName);

        sheetBuilder.TestSplitDataAcrossSheets(data);

        var sheetsField = typeof(ExcelFileBuilder).GetField("_sheets", BindingFlags.NonPublic | BindingFlags.Instance);
        Assert.That(sheetsField, Is.Not.Null, "Could not find _sheets field via reflection");
        
        var sheets = sheetsField.GetValue(realFileBuilder) as List<ISheetConfiguration>;
        Assert.That(sheets, Is.Not.Null);
        Assert.That(sheets, Has.Count.EqualTo(2), "Should have 2 additional sheets added to file builder");
        
        var firstAdditional = sheets[0] as SheetConfiguration<TestModel>;
        var secondAdditional = sheets[1] as SheetConfiguration<TestModel>;
        
        Assert.That(firstAdditional?.SheetName, Is.EqualTo($"{TestSheetName} 1"));
        Assert.That(firstAdditional?.Data, Has.Count.EqualTo(MaxRowsPerSheet));
        
        Assert.That(secondAdditional?.SheetName, Is.EqualTo($"{TestSheetName} 2"));
        Assert.That(secondAdditional?.Data, Has.Count.EqualTo(SmallDataSetSize));
    }

    [Test]
    public void SplitDataAcrossSheets_PreservesDataOrderAcrossAllSheets()
    {
        var data = GenerateTestData(LargeDataSetSize);
        var realFileBuilder = CreateRealFileBuilder();
        var sheetBuilder = new TestableSheetBuilder<TestModel>(realFileBuilder, TestSheetName);

        sheetBuilder.TestSplitDataAcrossSheets(data);
        var mainConfiguration = sheetBuilder.GetConfiguration();

        var sheetsField = typeof(ExcelFileBuilder).GetField("_sheets", BindingFlags.NonPublic | BindingFlags.Instance);
        var additionalSheets = sheetsField.GetValue(realFileBuilder) as List<ISheetConfiguration>;

        var allData = new List<TestModel>();
        allData.AddRange(mainConfiguration.Data);
        
        foreach (var sheet in additionalSheets)
        {
            var sheetConfig = sheet as SheetConfiguration<TestModel>;
            allData.AddRange(sheetConfig!.Data);
        }
        
        Assert.That(allData, Has.Count.EqualTo(data.Count), "Total data count should match");
        
        for (int i = 0; i < allData.Count; i++)
        {
            Assert.That(allData[i].Id, Is.EqualTo(data[i].Id), $"Data order mismatch at index {i}");
        }
    }

    [Test]
    public void CloneConfiguration_CreatesNewInstanceWithNewSheetName()
    {
        var newSheetName = "ClonedSheet";
        var testDataSize = 10;
        _sheetBuilder.GetConfiguration().Data = GenerateTestData(testDataSize);

        var cloned = _sheetBuilder.TestCloneConfiguration(newSheetName);

        Assert.That(cloned.SheetName, Is.EqualTo(newSheetName));
        Assert.That(cloned, Is.Not.SameAs(_sheetBuilder.GetConfiguration()));
    }

    [Test]
    public void CloneConfiguration_CopiesDropdownData()
    {
        var dropdownData = new List<ExcelDropdownData>
        {
            new ExcelDropdownData
            {
                ColumnName = "Status",
                DataList = new List<object> { "Active", "Inactive" },
                BindProperties = new List<string> { "StatusProperty" }
            }
        };
        
        _sheetBuilder.WithDropdownData(dropdownData, "DropdownSheet");
        
        var cloned = _sheetBuilder.TestCloneConfiguration("ClonedSheet");

        Assert.That(cloned.DropdownData, Is.EqualTo(dropdownData));
        Assert.That(cloned.DropdownSheetName, Is.EqualTo("DropdownSheet"));
    }

    [Test]
    public void CloneConfiguration_CopiesConditionalRules()
    {
        var conditionalRules = new Mock<IExcelConditionalRulesFactory<TestModel>>();
        conditionalRules.Setup(x => x.Rules).Returns(new List<ExcelConditionalRule>
        {
            new ExcelConditionalRule
            {
                SourcePropertyName = "Property1",
                TargetPropertyName = "Property2",
                Operator = ConditionalOperator.Equal,
                Values = new List<string> { "Test" },
                Actions = new List<ConditionalAction>
                {
                    new ConditionalAction { Type = ConditionalActionType.ChangeBackgroundColor, Value = "#FF0000" }
                }
            }
        });
        
        _sheetBuilder.WithConditionalRules(conditionalRules.Object);
        
        var cloned = _sheetBuilder.TestCloneConfiguration("ClonedSheet");

        Assert.That(cloned.ConditionalRules, Is.EqualTo(_sheetBuilder.GetConfiguration().ConditionalRules));
    }

    [Test]
    public void CloneConfiguration_CopiesExcludedProperties()
    {
        _sheetBuilder.ExcludeProperties(x => x.Property1, x => x.Property3);
        var originalExcluded = _sheetBuilder.GetConfiguration().ExcludedProperties.ToList();

        var cloned = _sheetBuilder.TestCloneConfiguration("ClonedSheet");

        Assert.That(cloned.ExcludedProperties, Is.EquivalentTo(originalExcluded));
    }

    [Test]
    public void CloneConfiguration_DoesNotCopyData()
    {
        var testDataSize = 100;
        _sheetBuilder.GetConfiguration().Data = GenerateTestData(testDataSize);

        var cloned = _sheetBuilder.TestCloneConfiguration("ClonedSheet");

        Assert.That(cloned.Data, Is.Empty, "Cloned configuration should not copy data");
    }

    [Test]
    public void CloneConfiguration_CreatesIndependentExcludedPropertiesSet()
    {
        _sheetBuilder.ExcludeProperties(x => x.Property1);
        var cloned = _sheetBuilder.TestCloneConfiguration("ClonedSheet");

        _sheetBuilder.GetConfiguration().ExcludedProperties.Add("NewProperty");

        Assert.That(cloned.ExcludedProperties.Contains("NewProperty"), Is.False, 
            "Cloned excluded properties should be independent");
    }

    private ExcelFileBuilder CreateRealFileBuilder()
    {
        var mockCacheManager = new Mock<IExcelCacheManager>();
        var mockCellWriter = new Mock<IExcelCellWriter>();
        var mockStyleManager = new Mock<IExcelStyleManager>();
        var mockValidationManager = new Mock<IExcelValidationManager>();
        var stringBuilderPool = new StringBuilderPool();
        var mockPropertyNameFormatter = new Mock<IPropertyNameFormatter>();
        var mockWorkbookWriter = new Mock<IExcelWorkbookWriter>();

        return new ExcelFileBuilder(
            mockCacheManager.Object,
            mockCellWriter.Object,
            mockStyleManager.Object,
            mockValidationManager.Object,
            stringBuilderPool,
            mockPropertyNameFormatter.Object,
            mockWorkbookWriter.Object
        );
    }

    private List<TestModel> GenerateTestData(int count)
    {
        var data = new List<TestModel>();
        for (int i = 0; i < count; i++)
        {
            data.Add(new TestModel
            {
                Id = i + 1,
                Property1 = $"Value1_{i}",
                Property2 = $"Value2_{i}",
                Property3 = i * TestDataMultiplier,
                Property4 = DateTime.Now.AddDays(i),
                Property5 = i % 2 == 0
            });
        }
        return data;
    }

    private List<LargeTestModel> GenerateLargeTestData(int count)
    {
        var data = new List<LargeTestModel>();
        for (int i = 0; i < count; i++)
        {
            data.Add(new LargeTestModel { Id = i + 1 });
        }
        return data;
    }

    private class TestableSheetBuilder<T> : SheetBuilder<T> where T : class, new()
    {
        public TestableSheetBuilder(ExcelFileBuilder fileBuilder, string sheetName) 
            : base(fileBuilder, sheetName)
        {
        }

        public bool TestShouldSplitData(List<T> data) => ShouldSplitData(data);
        public void TestSplitDataAcrossSheets(List<T> data) => SplitDataAcrossSheets(data);
        public SheetConfiguration<T> TestCloneConfiguration(string sheetName) => CloneConfiguration(sheetName);
        
        public SheetConfiguration<T> GetConfiguration()
        {
            var configField = typeof(SheetBuilder<T>).GetField("_configuration", BindingFlags.NonPublic | BindingFlags.Instance);
            return configField?.GetValue(this) as SheetConfiguration<T>;
        }
    }
}
