[Test]
[TestCase("A1", 0)]
[TestCase("B1", 1)]
[TestCase("Z1", 25)]
[TestCase("AA1", 26)]
[TestCase("AB1", 27)]
[TestCase("AZ1", 51)]
[TestCase("BA1", 52)]
[TestCase("ZZ1", 701)]
[TestCase("AAA1", 702)]
public void ReadSheet_ColumnReferences_MapsCorrectly(string cellReference, int expectedColumnIndex)
{
    // This indirectly tests the GetColumnIndex method by verifying column mapping
    // Note: This test validates the algorithm logic, actual data reading is tested in other tests
    
    // Create a test service instance to verify column mapping works correctly
    var testService = new TestableExcelReadService<ExcelReadTestModel>();
    var actualIndex = testService.TestGetColumnIndex(cellReference);
    
    Assert.That(actualIndex, Is.EqualTo(expectedColumnIndex), 
        $"Cell reference {cellReference} should map to column index {expectedColumnIndex}");
}

// Helper class to test protected methods
private class TestableExcelReadService<T> : ExcelReadService<T> where T : new()
{
    public int TestGetColumnIndex(string cellReference)
    {
        return base.GetColumnIndex(cellReference);
    }
    
    public string TestGetCellValue(Cell cell, Dictionary<string, string> sharedStrings)
    {
        return base.GetCellValue(cell, sharedStrings);
    }
}
