    // You may need to create a separate test file for this scenario
    var emptyPropertiesFileName = "Gui.Core.Excel.Tests.TestFiles.NoProperties.xlsx";
    using var stream = Assembly.GetExecutingAssembly().GetManifestResourceStream(emptyPropertiesFileName);
    
    if (stream != null)
    {
        var properties = _service.ReadFileProperties(stream);
        
        Assert.That(properties, Is.Not.Null);
        // Properties should be null or empty when not set in the file
        Assert.That(properties.Title, Is.Null.Or.Empty);

        Assert.That(properties.Keywords, Is.Null.Or.Empty);
