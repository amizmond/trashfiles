using DocumentFormat.OpenXml.Packaging;
using DynamicExcel.Write;
using DynamicExcel.Write.Conditional;
using DynamicExcel.Write.Helpers;
using DynamicExcel.Write.Models;
using DynamicExcel.Write.Sensitivity;
using Moq;
using System.Linq.Expressions;

namespace DynamicExcel.Tests;

[TestFixture]
public class ExcelFileBuilderTests
{
    private Mock<IExcelCacheManager> _cacheManagerMock;
    private Mock<IExcelCellWriter> _cellWriterMock;
    private Mock<IExcelStyleManager> _styleManagerMock;
    private Mock<IExcelValidationManager> _validationManagerMock;
    private StringBuilderPool _stringBuilderPool;
    private Mock<IPropertyNameFormatter> _propertyNameFormatterMock;
    private Mock<IExcelWorkbookWriter> _workbookWriterMock;
    private Mock<IExcelSensitivityManager> _sensitivityManagerMock;
    private ExcelFileBuilder _builder;

    [SetUp]
    public void Setup()
    {
        _cacheManagerMock = new Mock<IExcelCacheManager>();
        _cellWriterMock = new Mock<IExcelCellWriter>();
        _styleManagerMock = new Mock<IExcelStyleManager>();
        _validationManagerMock = new Mock<IExcelValidationManager>();
        _stringBuilderPool = new StringBuilderPool();
        _propertyNameFormatterMock = new Mock<IPropertyNameFormatter>();
        _workbookWriterMock = new Mock<IExcelWorkbookWriter>();
        _sensitivityManagerMock = new Mock<IExcelSensitivityManager>();

        _builder = new ExcelFileBuilder(
            _cacheManagerMock.Object,
            _cellWriterMock.Object,
            _styleManagerMock.Object,
            _validationManagerMock.Object,
            _stringBuilderPool,
            _propertyNameFormatterMock.Object,
            _workbookWriterMock.Object,
            _sensitivityManagerMock.Object);
    }

    [Test]
    public void Constructor_WithAllDependencies_CreatesInstance()
    {
        var builder = new ExcelFileBuilder(
            _cacheManagerMock.Object,
            _cellWriterMock.Object,
            _styleManagerMock.Object,
            _validationManagerMock.Object,
            _stringBuilderPool,
            _propertyNameFormatterMock.Object,
            _workbookWriterMock.Object,
            _sensitivityManagerMock.Object);

        Assert.That(builder, Is.Not.Null);
    }

    [Test]
    public void Constructor_WithNullCacheManager_ThrowsArgumentNullException()
    {
        Assert.Throws<ArgumentNullException>(() =>
        {
            var excelFileBuilder = new ExcelFileBuilder(
                null,
                _cellWriterMock.Object,
                _styleManagerMock.Object,
                _validationManagerMock.Object,
                new StringBuilderPool(),
                _propertyNameFormatterMock.Object,
                _workbookWriterMock.Object,
                _sensitivityManagerMock.Object);
        });
    }

    [Test]
    public void Constructor_Parameterless_CreatesInstanceWithDefaults()
    {
        var builder = new ExcelFileBuilder();
        Assert.That(builder, Is.Not.Null);
    }

    [Test]
    public void AddSheet_WithValidSheetName_ReturnsSheetBuilder()
    {
        const string sheetName = "TestSheet";
        var sheetBuilder = _builder.AddSheet<TestModel>(sheetName);
        Assert.That(sheetBuilder, Is.Not.Null);
    }

    [Test]
    public void AddSheet_WithNullSheetName_ThrowsArgumentException()
    {
        Assert.Throws<ArgumentNullException>(() => _builder.AddSheet<TestModel>(null));
    }

    [Test]
    public void AddSheet_WithEmptySheetName_ThrowsArgumentException()
    {
        Assert.Throws<ArgumentException>(() => _builder.AddSheet<TestModel>(string.Empty));
    }

    [Test]
    public void AddSimpleSheet_WithValidData_AddsSheetConfiguration()
    {
        var data = new List<TestModel> { new TestModel { Id = 1, Name = "Test" } };
        _propertyNameFormatterMock.Setup(x => x.ConvertToFriendlyName(It.IsAny<string>()))
            .Returns("Test Model");

        var result = _builder.AddSimpleSheet(data, "SimpleSheet");

        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void AddSimpleSheet_WithNullData_ThrowsArgumentNullException()
    {
        Assert.Throws<ArgumentNullException>(() => _builder.AddSimpleSheet<TestModel>(null));
    }

    [Test]
    public void AddSimpleSheet_WithNullSheetName_GeneratesName()
    {
        var data = new List<TestModel> { new TestModel() };
        _propertyNameFormatterMock.Setup(x => x.ConvertToFriendlyName("Test")).Returns("Test");

        var result = _builder.AddSimpleSheet(data);

        Assert.That(result, Is.EqualTo(_builder));
        _propertyNameFormatterMock.Verify(x => x.ConvertToFriendlyName("Test"), Times.Once);
    }

    [Test]
    public void WithSensitivity_NoParameters_SetsDefaultInternal()
    {
        var result = _builder.WithSensitivity();
        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void WithSensitivity_WithLevel_SetsSpecifiedLevel()
    {
        var result = _builder.WithSensitivity(SensitivityLevel.Confidential);
        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void WithSensitivity_WithConfiguration_AppliesConfiguration()
    {
        var result = _builder.WithSensitivity(config => { config.Level = SensitivityLevel.Public; });
        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void WithSensitivity_WithNullConfiguration_ThrowsArgumentNullException()
    {
        Assert.Throws<ArgumentNullException>(() => _builder.WithSensitivity(null));
    }

    [Test]
    public void WithFileProperties_ValidConfiguration_AppliesProperties()
    {
        var result = _builder.WithFileProperties(props =>
        {
            props.Title = "Test Title";
            props.Subject = "Test Subject";
            props.Creator = "Test Creator";
        });

        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void WithFileProperties_NullConfiguration_ThrowsArgumentNullException()
    {
        Assert.Throws<ArgumentNullException>(() => _builder.WithFileProperties(null));
    }

    [Test]
    public void Build_WithSimpleSheet_CreatesExcelFile()
    {
        var data = new List<TestModel> { new TestModel { Id = 1, Name = "Test" } };
        var propertyAccessors = CreateMockPropertyAccessors();

        SetupMocksForBuild(propertyAccessors);

        _builder.AddSimpleSheet(data, "TestSheet");

        var result = _builder.Build();

        Assert.That(result, Is.Not.Null);
        Assert.That(result, Is.TypeOf<MemoryStream>());
        Assert.That(result.Position, Is.EqualTo(0));

        result.Dispose();
    }

    [Test]
    public void SheetBuilder_WithData_ConfiguresSheet()
    {
        var data = new List<TestModel> { new TestModel { Id = 1, Name = "Test" } };

        var result = _builder
            .AddSheet<TestModel>("TestSheet")
            .WithData(data)
            .Done();

        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void SheetBuilder_ExcludeProperties_ExcludesSpecifiedProperties()
    {
        Expression<Func<TestModel, object>> excludeExpression = x => x.Name;

        var result = _builder
            .AddSheet<TestModel>("TestSheet")
            .ExcludeProperties(excludeExpression)
            .Done();

        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void SheetBuilder_WithDropdownData_ConfiguresDropdowns()
    {
        var dropdownData = new List<ExcelDropdownData>
        {
            new ExcelDropdownData
            {
                ColumnName = "Status",
                DataList = new List<object> { "Active", "Inactive" },
                BindProperties = new List<string> { "Status" },
            }
        };

        var result = _builder
            .AddSheet<TestModel>("TestSheet")
            .WithDropdownData(dropdownData)
            .Done();

        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void SheetBuilder_WithConditionalRules_ConfiguresRules()
    {
        var conditionalFactory = new Mock<IExcelConditionalRulesFactory<TestModel>>();
        conditionalFactory.Setup(x => x.Rules).Returns(new List<ExcelConditionalRule>());

        var result = _builder
            .AddSheet<TestModel>("TestSheet")
            .WithConditionalRules(conditionalFactory.Object)
            .Done();

        Assert.That(result, Is.EqualTo(_builder));
    }

    [Test]
    public void Build_WithMultipleSheets_CreatesAllSheets()
    {
        var data1 = new List<TestModel> { new TestModel { Id = 1 } };
        var data2 = new List<TestModel> { new TestModel { Id = 2 } };
        var propertyAccessors = CreateMockPropertyAccessors();

        SetupMocksForBuild(propertyAccessors);

        _builder
            .AddSimpleSheet(data1, "Sheet1")
            .AddSimpleSheet(data2, "Sheet2");

        var result = _builder.Build();

        Assert.That(result, Is.Not.Null);
        Assert.That(result.Length, Is.GreaterThan(0));

        result.Dispose();
    }

    private PropertyAccessor[] CreateMockPropertyAccessors()
    {
        return new[]
        {
            new PropertyAccessor
            {
                Property = typeof(TestModel).GetProperty(nameof(TestModel.Id)),
                ColumnName = "Id",
                Order = 0,
                Width = 15,
                Getter = obj => ((TestModel)obj).Id,
                PropertyType = typeof(int),
                UnderlyingType = typeof(int),
                IsNullable = false,
                IsReadOnly = false,
            },
            new PropertyAccessor
            {
                Property = typeof(TestModel).GetProperty(nameof(TestModel.Name)),
                ColumnName = "Name",
                Order = 1,
                Width = 15,
                Getter = obj => ((TestModel)obj).Name,
                PropertyType = typeof(string),
                UnderlyingType = typeof(string),
                IsNullable = true,
                IsReadOnly = false,
            }
        };
    }

    private void SetupMocksForBuild(PropertyAccessor[] propertyAccessors)
    {
        _cacheManagerMock.Setup(x => x.GetPropertyAccessors<TestModel>(It.IsAny<HashSet<string>>()))
            .Returns(propertyAccessors);

        _cacheManagerMock.Setup(x => x.GetExcelColumnName(It.IsAny<int>()))
            .Returns<int>(col => col switch
            {
                1 => "A",
                2 => "B",
                _ => "C",
            });

        _cacheManagerMock.Setup(x => x.GetCleanHexColor(It.IsAny<string>()))
            .Returns<string>(color => color?.Replace("#", ""));

        _styleManagerMock.Setup(x => x.CreateStyles(
                It.IsAny<WorkbookPart>(),
                It.IsAny<PropertyAccessor[]>(),
                It.IsAny<CancellationToken>()))
            .Returns(new Dictionary<string, uint> { { "test", 1 } });

        _propertyNameFormatterMock.Setup(x => x.ConvertToFriendlyName(It.IsAny<string>()))
            .Returns<string>(name => name);
    }

    public class TestModel
    {
        [ExcelColumn("Id", 0)]
        public int Id { get; set; }

        [ExcelColumn("Name", 1)]
        public string Name { get; set; }

        [ExcelColumn("Status", 2)]
        public string Status { get; set; }

        [ExcelColumn("Date", 3)]
        public DateTime Date { get; set; }

        [ExcelColumn("Amount", 4)]
        public decimal Amount { get; set; }
    }
}
