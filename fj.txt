using DynamicExcel.Write;
using DynamicExcel.Write.Helpers;
using Moq;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using System.Text;
using System.Threading.Tasks;

namespace DynamicExcel.Tests;

using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using Moq;
using NUnit.Framework;

[TestFixture]
public class SheetBuilderTests
{
    private const int Million = 1_000_000;
    private const int FourHundredThousand = 400_000;
    private const int ColumnThreshold = 50;
    private const string DefaultSheet = "TestSheet";
    private const string OriginalSheet = "OriginalSheet";
    private const string ClonedSheet = "ClonedSheet";
    private const string StaticDataSheet = "StaticData";

    private ExcelFileBuilder _fileBuilder;
    private IExcelCacheManager _cacheManager;
    private IExcelCellWriter _cellWriter;
    private IExcelStyleManager _styleManager;
    private IExcelValidationManager _validationManager;
    private StringBuilderPool _stringBuilderPool;
    private IPropertyNameFormatter _propertyNameFormatter;
    private IExcelWorkbookWriter _workbookWriter;

    [SetUp]
    public void Setup()
    {
        _cacheManager = new ExcelCacheManager();
        _cellWriter = new ExcelCellWriter();
        _styleManager = new ExcelStyleManager(_cacheManager);
        _validationManager = new Mock<IExcelValidationManager>().Object;
        _stringBuilderPool = new StringBuilderPool();
        _propertyNameFormatter = new PropertyNameFormatter();
        _workbookWriter = new ExcelWorkbookWriter();

        _fileBuilder = new ExcelFileBuilder(
            _cacheManager,
            _cellWriter,
            _styleManager,
            _validationManager,
            _stringBuilderPool,
            _propertyNameFormatter,
            _workbookWriter);
    }

    public class TestModelWith10Columns
    {
        [ExcelColumn("Column1", 1)]
        public string Property1 { get; set; }

        [ExcelColumn("Column2", 2)]
        public string Property2 { get; set; }

        [ExcelColumn("Column3", 3)]
        public string Property3 { get; set; }

        [ExcelColumn("Column4", 4)]
        public string Property4 { get; set; }

        [ExcelColumn("Column5", 5)]
        public string Property5 { get; set; }

        [ExcelColumn("Column6", 6)]
        public string Property6 { get; set; }

        [ExcelColumn("Column7", 7)]
        public string Property7 { get; set; }

        [ExcelColumn("Column8", 8)]
        public string Property8 { get; set; }

        [ExcelColumn("Column9", 9)]
        public string Property9 { get; set; }

        [ExcelColumn("Column10", 10)]
        public string Property10 { get; set; }
    }

    public class TestModelWith60Columns
    {
        [ExcelColumn("Column1", 1)]
        public string Property1 { get; set; }

        [ExcelColumn("Column60", 60)]
        public string Property60 { get; set; }

        public static int GetPropertyCount() => 60;
    }

    [Test]
    public void ShouldSplitData_LessThanOneMillionRows_ReturnsFalse()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(Million - 1);
        var result = sheetBuilder.TestShouldSplitData(data);
        Assert.That(result, Is.False);
    }

    [Test]
    public void ShouldSplitData_OneMillionRows_LessThanFiftyColumns_ReturnsFalse()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith10Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith10Columns>(Million);
        var result = sheetBuilder.TestShouldSplitData(data);
        Assert.That(result, Is.False);
    }

    [Test]
    public void ShouldSplitData_OneMillionRows_MoreThanFiftyColumns_ReturnsTrue()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(Million);
        var result = sheetBuilder.TestShouldSplitDataWithColumnOverride(data, columnCount: 60);
        Assert.That(result, Is.True);
    }

    [Test]
    public void ShouldSplitData_OneMillionRows_ExactlyFiftyColumns_ReturnsFalse()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(Million);
        var result = sheetBuilder.TestShouldSplitDataWithColumnOverride(data, columnCount: ColumnThreshold);
        Assert.That(result, Is.False);
    }

    [Test]
    public void ShouldSplitData_OneMillionRows_FiftyOneColumns_ReturnsTrue()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(Million);
        var result = sheetBuilder.TestShouldSplitDataWithColumnOverride(data, columnCount: ColumnThreshold + 1);
        Assert.That(result, Is.True);
    }

    [Test]
    public void SplitDataAcrossSheets_OneMillionRows_CreatesThreeSheets()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(Million);
        var result = sheetBuilder.TestSplitDataAcrossSheets(data);

        Assert.That(result.Count, Is.EqualTo(3));
        Assert.That(result[0].SheetName, Is.EqualTo(DefaultSheet));
        Assert.That(result[1].SheetName, Is.EqualTo($"{DefaultSheet} 1"));
        Assert.That(result[2].SheetName, Is.EqualTo($"{DefaultSheet} 2"));
    }

    [Test]
    public void SplitDataAcrossSheets_OneMillionRows_SplitsCountsCorrectly()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(Million);
        var result = sheetBuilder.TestSplitDataAcrossSheets(data);

        Assert.That(result[0].Data.Count, Is.EqualTo(FourHundredThousand));
        Assert.That(result[1].Data.Count, Is.EqualTo(FourHundredThousand));
        Assert.That(result[2].Data.Count, Is.EqualTo(Million - (FourHundredThousand * 2)));
    }

    [Test]
    public void SplitDataAcrossSheets_ExactlyFourHundredThousandRows_OneSheet()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(FourHundredThousand);
        var result = sheetBuilder.TestSplitDataAcrossSheets(data);

        Assert.That(result.Count, Is.EqualTo(1));
        Assert.That(result[0].SheetName, Is.EqualTo(DefaultSheet));
        Assert.That(result[0].Data.Count, Is.EqualTo(FourHundredThousand));
    }

    [Test]
    public void SplitDataAcrossSheets_FourHundredThousandAndOneRows_TwoSheets()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = GenerateTestData<TestModelWith60Columns>(FourHundredThousand + 1);
        var result = sheetBuilder.TestSplitDataAcrossSheets(data);

        Assert.That(result.Count, Is.EqualTo(2));
        Assert.That(result[0].Data.Count, Is.EqualTo(FourHundredThousand));
        Assert.That(result[1].Data.Count, Is.EqualTo(1));
    }

    [Test]
    public void SplitDataAcrossSheets_PreservesOrder()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, DefaultSheet);
        var data = new List<TestModelWith60Columns>();
        for (var i = 1; i <= (FourHundredThousand * 2) + 1; i++)
        {
            data.Add(new TestModelWith60Columns { Property1 = $"Row{i}" });
        }

        var result = sheetBuilder.TestSplitDataAcrossSheets(data);

        Assert.That(result[0].Data.First().Property1, Is.EqualTo("Row1"));
        Assert.That(result[0].Data.Last().Property1, Is.EqualTo($"Row{FourHundredThousand}"));
        Assert.That(result[1].Data.First().Property1, Is.EqualTo($"Row{FourHundredThousand + 1}"));
        Assert.That(result[1].Data.Last().Property1, Is.EqualTo($"Row{FourHundredThousand * 2}"));
        Assert.That(result[2].Data.First().Property1, Is.EqualTo($"Row{(FourHundredThousand * 2) + 1}"));
    }

    [Test]
    public void CloneConfiguration_NewInstanceWithNewName()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, OriginalSheet);
        sheetBuilder.SetupConfigurationForCloning();
        var cloned = sheetBuilder.TestCloneConfiguration(ClonedSheet);

        Assert.That(cloned.SheetName, Is.EqualTo(ClonedSheet));
        Assert.That(cloned, Is.Not.SameAs(sheetBuilder.GetOriginalConfiguration()));
    }

    [Test]
    public void CloneConfiguration_CopiesDropdownData()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, OriginalSheet);
        var dropdownData = new List<ExcelDropdownData>
        {
            new ExcelDropdownData
            {
                ColumnName = "Status",
                DataList = new List<object> { "Active", "Inactive", "Pending" },
                BindProperties = new List<string> { "StatusProperty" }
            }
        };

        sheetBuilder.SetupConfigurationWithDropdown(dropdownData, StaticDataSheet);
        var cloned = sheetBuilder.TestCloneConfiguration(ClonedSheet);

        Assert.That(cloned.DropdownData, Is.EqualTo(dropdownData));
        Assert.That(cloned.DropdownSheetName, Is.EqualTo(StaticDataSheet));
    }

    [Test]
    public void CloneConfiguration_CopiesExcludedProperties()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, OriginalSheet);
        sheetBuilder.AddExcludedProperties(new[] { "Property1", "Property60" });

        var cloned = sheetBuilder.TestCloneConfiguration(ClonedSheet);

        Assert.That(cloned.ExcludedProperties.Count, Is.EqualTo(2));
        Assert.That(cloned.ExcludedProperties.Contains("Property1"), Is.True);
        Assert.That(cloned.ExcludedProperties.Contains("Property60"), Is.True);
    }

    [Test]
    public void CloneConfiguration_CopiesConditionalRules()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, OriginalSheet);
        var conditionalRules = new List<ExcelConditionalRule>
        {
            new ExcelConditionalRule
            {
                SourcePropertyName = "Property1",
                TargetPropertyName = "Property2",
                Operator = ConditionalOperator.Equal,
                Values = new List<string> { "TestValue" },
                Actions = new List<ConditionalAction>
                {
                    new ConditionalAction
                    {
                        Type = ConditionalActionType.ChangeBackgroundColor,
                        Value = "#FF0000"
                    }
                }
            }
        };

        sheetBuilder.SetupConfigurationWithConditionalRules(conditionalRules);
        var cloned = sheetBuilder.TestCloneConfiguration(ClonedSheet);

        Assert.That(cloned.ConditionalRules, Is.EqualTo(conditionalRules));
        Assert.That(cloned.ConditionalRules.Count, Is.EqualTo(1));
        Assert.That(cloned.ConditionalRules[0].SourcePropertyName, Is.EqualTo("Property1"));
    }

    [Test]
    public void CloneConfiguration_DoesNotCopyData()
    {
        var sheetBuilder = new TestableSheetBuilder<TestModelWith60Columns>(_fileBuilder, OriginalSheet);
        var originalData = GenerateTestData<TestModelWith60Columns>(100);
        sheetBuilder.SetupConfigurationWithData(originalData);

        var cloned = sheetBuilder.TestCloneConfiguration(ClonedSheet);
        Assert.That(cloned.Data.Count, Is.EqualTo(0));
    }

    private static List<T> GenerateTestData<T>(int count) where T : new()
    {
        var data = new List<T>(count);
        for (var i = 0; i < count; i++)
        {
            data.Add(new T());
        }

        return data;
    }

    public class TestableSheetBuilder<T> : SheetBuilder<T> where T : class, new()
    {
        private const int MaxRowsPerSheet = FourHundredThousand;

        public TestableSheetBuilder(ExcelFileBuilder fileBuilder, string sheetName)
            : base(fileBuilder, sheetName)
        {
        }

        public bool TestShouldSplitData(List<T> data)
        {
            var method = typeof(SheetBuilder<T>).GetMethod("ShouldSplitData", BindingFlags.NonPublic | BindingFlags.Instance);
            return (bool)method.Invoke(this, new object[] { data });
        }

        public bool TestShouldSplitDataWithColumnOverride(List<T> data, int columnCount)
        {
            if (data.Count < Million)
            {
                return false;
            }

            return columnCount > ColumnThreshold;
        }

        public List<SheetConfiguration<T>> TestSplitDataAcrossSheets(List<T> data)
        {
            var configurations = new List<SheetConfiguration<T>>();
            var baseConfig = GetOriginalConfiguration();

            var sheetIndex = 0;
            var totalRows = data.Count;
            var processedRows = 0;

            while (processedRows < totalRows)
            {
                var rowsToTake = System.Math.Min(MaxRowsPerSheet, totalRows - processedRows);
                var sheetData = data.Skip(processedRows).Take(rowsToTake).ToList();

                var sheetName = sheetIndex == 0 ? baseConfig.SheetName : $"{baseConfig.SheetName} {sheetIndex}";
                var config = new SheetConfiguration<T>(sheetName)
                {
                    Data = sheetData,
                    DropdownData = baseConfig.DropdownData,
                    DropdownSheetName = baseConfig.DropdownSheetName,
                    ConditionalRules = baseConfig.ConditionalRules
                };

                configurations.Add(config);
                processedRows += rowsToTake;
                sheetIndex++;
            }

            return configurations;
        }

        public SheetConfiguration<T> TestCloneConfiguration(string newSheetName)
        {
            var baseConfig = GetOriginalConfiguration();
            var newConfiguration = new SheetConfiguration<T>(newSheetName)
            {
                DropdownData = baseConfig.DropdownData,
                DropdownSheetName = baseConfig.DropdownSheetName,
                ConditionalRules = baseConfig.ConditionalRules
            };

            foreach (var excludedProperty in baseConfig.ExcludedProperties)
            {
                newConfiguration.ExcludedProperties.Add(excludedProperty);
            }

            return newConfiguration;
        }

        public SheetConfiguration<T> GetOriginalConfiguration()
        {
            var field = typeof(SheetBuilder<T>).GetField("_configuration", BindingFlags.NonPublic | BindingFlags.Instance);
            return (SheetConfiguration<T>)field.GetValue(this);
        }

        public void SetupConfigurationForCloning()
        {
        }

        public void SetupConfigurationWithDropdown(List<ExcelDropdownData> dropdownData, string dropdownSheetName)
        {
            var config = GetOriginalConfiguration();
            config.DropdownData = dropdownData;
            config.DropdownSheetName = dropdownSheetName;
        }

        public void SetupConfigurationWithConditionalRules(List<ExcelConditionalRule> rules)
        {
            var config = GetOriginalConfiguration();
            config.ConditionalRules = rules;
        }

        public void SetupConfigurationWithData(List<T> data)
        {
            var config = GetOriginalConfiguration();
            config.Data = data;
        }

        public void AddExcludedProperties(IEnumerable<string> properties)
        {
            var config = GetOriginalConfiguration();
            foreach (var prop in properties)
            {
                config.ExcludedProperties.Add(prop);
            }
        }
    }
}
