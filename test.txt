================================================================================
           IFRAME-BASED ANGULAR PORTAL - STEP-BY-STEP GUIDE
================================================================================

INTRODUCTION
------------
This guide walks you through creating an iframe-based portal that serves as an
entry point for multiple Angular applications across different environments.

What You Will Build:
- A home page displaying all available applications
- Environment selector (Development, Staging, Production)
- Application switcher in the header
- Iframe container for embedded applications
- Communication service for portal-to-app messaging

Prerequisites:
- Node.js 18+ installed
- Angular CLI 17+ installed globally
- Basic knowledge of Angular and TypeScript


================================================================================
STEP 1: CREATE NEW ANGULAR PROJECT
================================================================================

Run the following commands:

    ng new app-portal --standalone --routing --style=scss
    cd app-portal

When prompted for SSR/SSG, select NO (not needed for this portal).

        { name: 'Staging', url: 'https://staging-inventory.yourcompany.com' },
        { name: 'Production', url: 'https://inventory.yourcompany.com' }
      ]
    },
    {
      id: 'reports',
      name: 'Reports & Analytics',
      description: 'Business intelligence dashboard',
      icon: '📊',
      color: '#00aeef',
      environments: [
        { name: 'Development', url: 'https://dev-reports.yourcompany.com' },
        { name: 'Staging', url: 'https://staging-reports.yourcompany.com' },
        { name: 'Production', url: 'https://reports.yourcompany.com' }
      ]
    }
  ];

  // Signals for reactive state
  private portalState = signal<PortalState>({
    currentApp: null,
    currentEnvironment: null,
    isLoading: false
  });

  // Computed values
  readonly state = this.portalState.asReadonly();
  readonly currentApp = computed(() => this.portalState().currentApp);
  readonly currentEnvironment = computed(() => this.portalState().currentEnvironment);
  readonly isLoading = computed(() => this.portalState().isLoading);
  readonly currentUrl = computed(() => this.portalState().currentEnvironment?.url || null);

  getApplications(): AppConfig[] {
    return this.applications;
  }

  getAppById(id: string): AppConfig | undefined {
    return this.applications.find(app => app.id === id);
  }

  selectApp(appId: string, environmentIndex: number = 0): void {
    const app = this.getAppById(appId);
    if (app) {
      this.portalState.update(state => ({
        ...state,
        currentApp: app,
        currentEnvironment: app.environments[environmentIndex],
        isLoading: true
      }));
    }
  }

  selectEnvironment(environment: Environment): void {
    this.portalState.update(state => ({
      ...state,
      currentEnvironment: environment,
      isLoading: true
    }));
  }

  setLoading(isLoading: boolean): void {
    this.portalState.update(state => ({ ...state, isLoading }));
  }

  goHome(): void {
    this.portalState.set({
      currentApp: null,
      currentEnvironment: null,
      isLoading: false
    });
  }
}

import { Injectable, NgZone } from '@angular/core';
import { Subject, Observable, filter } from 'rxjs';

export interface PortalMessage {
  type: string;
  payload?: any;
  source?: string;
  timestamp?: number;
}

@Injectable({ providedIn: 'root' })
export class PortalCommunicationService {
  private messageSubject = new Subject<PortalMessage>();
  private iframeElement: HTMLIFrameElement | null = null;

  constructor(private ngZone: NgZone) {
    this.initMessageListener();
  }

  private initMessageListener(): void {
    window.addEventListener('message', (event: MessageEvent) => {
      this.ngZone.run(() => {
        if (event.data && event.data.type) {
          this.messageSubject.next({
            ...event.data,
            timestamp: Date.now()
          });
        }
      });
    });
  }

  setIframe(iframe: HTMLIFrameElement): void {
    this.iframeElement = iframe;
  }

  sendToApp(message: PortalMessage): void {
    if (this.iframeElement?.contentWindow) {
      this.iframeElement.contentWindow.postMessage(
        { ...message, source: 'portal', timestamp: Date.now() },
        '*'
      );
    }
  }

  onMessage(type?: string): Observable<PortalMessage> {
    if (type) {
      return this.messageSubject.pipe(filter(msg => msg.type === type));
    }
    return this.messageSubject.asObservable();
  }

  sendUserInfo(user: any): void {
    this.sendToApp({ type: 'USER_INFO', payload: user });
  }

  sendAuthToken(token: string): void {
    this.sendToApp({ type: 'AUTH_TOKEN', payload: token });
  }
}

export class PortalComponent {
  @ViewChild('appFrame') appFrame!: ElementRef<HTMLIFrameElement>;

  private registry = inject(AppRegistryService);
  private router = inject(Router);
  private sanitizer = inject(DomSanitizer);
  private communication = inject(PortalCommunicationService);

  applications = this.registry.getApplications();
  currentApp = this.registry.currentApp;
  currentEnvironment = this.registry.currentEnvironment;
  isLoading = this.registry.isLoading;
  
  safeUrl: SafeResourceUrl | null = null;

  constructor() {
    if (!this.currentApp()) {
      this.router.navigate(['/']);
      return;
    }
    this.updateSafeUrl();
  }

  private updateSafeUrl(): void {
    const url = this.registry.currentUrl();
    if (url) {
      this.safeUrl = this.sanitizer.bypassSecurityTrustResourceUrl(url);
    }
  }

  onAppChange(event: Event): void {
    const appId = (event.target as HTMLSelectElement).value;
    this.registry.selectApp(appId);
    this.updateSafeUrl();
  }

  onEnvChange(event: Event): void {
    const url = (event.target as HTMLSelectElement).value;
    const env = this.currentApp()?.environments.find(e => e.url === url);
    if (env) {
      this.registry.selectEnvironment(env);
      this.updateSafeUrl();
    }
  }

  onIframeLoad(): void {
    this.registry.setLoading(false);
    if (this.appFrame) {
      this.communication.setIframe(this.appFrame.nativeElement);
    }
  }

  refreshApp(): void {
    this.registry.setLoading(true);
    const currentUrl = this.registry.currentUrl();
    this.safeUrl = null;
    setTimeout(() => {
      if (currentUrl) {
        this.safeUrl = this.sanitizer.bypassSecurityTrustResourceUrl(currentUrl);
      }
    }, 100);
  }

  openInNewTab(): void {
    const url = this.registry.currentUrl();
    if (url) window.open(url, '_blank');
  }

  goHome(): void {
    this.registry.goHome();
    this.router.navigate(['/']);
  }
}

import { Injectable, NgZone } from '@angular/core';
import { Subject, Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class PortalBridgeService {
  private messageSubject = new Subject<any>();
  private isInPortal = window.parent !== window;

  constructor(private ngZone: NgZone) {
    window.addEventListener('message', (event) => {
      this.ngZone.run(() => {
        if (event.data?.source === 'portal') {
          this.messageSubject.next(event.data);
        }
      });
    });
  }

  sendToPortal(type: string, payload?: any): void {
    if (this.isInPortal) {
      window.parent.postMessage({ type, payload, source: 'app' }, '*');
    }
  }

  notifyReady(): void {
    this.sendToPortal('APP_READY');
  }

  onMessage(): Observable<any> {
    return this.messageSubject.asObservable();
  }

  isRunningInPortal(): boolean {
    return this.isInPortal;
  }
}
